# TASK: Implement API Capture File Storage
# 任务：实现 API 捕获文件存储

## Priority: 🟡 HIGH

## Context
我们刚修复了内存泄漏问题。现在需要实现文件存储功能，将 API 消息持久化到磁盘。

## Goal
将每个 Agent 的 API 请求/响应记录写入独立的 JSONL 文件，按日期分割。

## File Structure to Create
```
captures/
├── agent-1/
│   ├── 2026-03-15.jsonl      # Today's records
│   ├── 2026-03-14.jsonl      # Yesterday's records
│   └── ...
├── agent-2/
│   └── 2026-03-15.jsonl
├── agent-3/
│   └── ...
... (agent-4 to agent-30)
└── _metadata/
    └── last-id.json          # Track last capture ID
```

## Implementation

### Step 1: Create CaptureFileWriter Class

Create new file: `src/services/capture-file-writer.js`
```javascript
/**
 * @file services/capture-file-writer.js
 * @description Writes API captures to JSONL files, one per agent per day
 *              将 API 捕获写入 JSONL 文件，每个 Agent 每天一个文件
 * 
 * @ai-context
 * Key design principles:
 * 关键设计原则：
 * 1. Non-blocking async writes (不阻塞的异步写入)
 * 2. Per-agent file isolation (每 Agent 文件隔离)
 * 3. Daily file rotation (每日文件轮转)
 * 4. Buffer writes for performance (缓冲写入以提高性能)
 */

const fs = require('fs');
const path = require('path');

class CaptureFileWriter {
  constructor(options = {}) {
    // Base directory for capture files
    // 捕获文件的基础目录
    this.baseDir = options.baseDir || path.join(process.cwd(), 'captures');
    
    // Flush interval in ms (default: 1 second)
    // 刷新间隔毫秒（默认：1秒）
    this.flushInterval = options.flushInterval || 1000;
    
    // Write buffer (batch writes for performance)
    // 写入缓冲（批量写入以提高性能）
    this.writeBuffer = [];
    
    // Open file streams (reuse for efficiency)
    // 打开的文件流（复用以提高效率）
    this.streams = new Map(); // key: "agent-date", value: WriteStream
    
    // Current date string (for file rotation)
    // 当前日期字符串（用于文件轮转）
    this.currentDate = this.getDateString();
    
    // Flush timer
    // 刷新定时器
    this.flushTimer = null;
    
    // Initialized flag
    // 初始化标志
    this.initialized = false;
  }

  /**
   * Initialize the writer
   * 初始化写入器
   */
  async initialize() {
    if (this.initialized) return;
    
    console.log(`[CaptureFileWriter] Initializing with baseDir: ${this.baseDir}`);
    
    // Create base directory if not exists
    // 如果不存在则创建基础目录
    await fs.promises.mkdir(this.baseDir, { recursive: true });
    await fs.promises.mkdir(path.join(this.baseDir, '_metadata'), { recursive: true });
    
    // Start flush timer
    // 启动刷新定时器
    this.startFlushTimer();
    
    // Start date rotation checker
    // 启动日期轮转检查器
    this.startDateRotationChecker();
    
    this.initialized = true;
    console.log('[CaptureFileWriter] Initialized successfully');
  }

  /**
   * Write a capture record (non-blocking, buffered)
   * 写入捕获记录（非阻塞，缓冲）
   * 
   * @param {string} agentId - Agent identifier
   * @param {object} capture - Capture record to write
   */
  write(agentId, capture) {
    // Add to buffer (will be flushed periodically)
    // 添加到缓冲（将定期刷新）
    this.writeBuffer.push({
      agentId,
      record: {
        id: capture.id,
        ts: capture.ts || new Date().toISOString(),
        agent: agentId,
        provider: capture.provider,
        model: capture.model,
        req: {
          method: capture.req?.method || 'POST',
          path: capture.req?.path || '/v1/chat/completions',
          tokens: capture.req?.tokens || 0,
          // Only store truncated preview (already truncated by memory fix)
          // 只存储截断的预览（已被内存修复截断）
          bodyPreview: capture.req?.bodyPreview,
        },
        res: capture.res ? {
          status: capture.res.status,
          tokens: capture.res.tokens || 0,
          latency: capture.res.latency || 0,
          textPreview: capture.res.textPreview,
        } : null,
        cost: capture.cost || 0,
        error: capture.error,
      },
    });
  }

  /**
   * Flush buffer to files
   * 将缓冲刷新到文件
   */
  async flush() {
    if (this.writeBuffer.length === 0) return;
    
    // Take current buffer and reset
    // 取出当前缓冲并重置
    const toWrite = this.writeBuffer;
    this.writeBuffer = [];
    
    // Group by agent
    // 按 Agent 分组
    const byAgent = new Map();
    for (const item of toWrite) {
      const existing = byAgent.get(item.agentId) || [];
      existing.push(item.record);
      byAgent.set(item.agentId, existing);
    }
    
    // Write each agent's records
    // 写入每个 Agent 的记录
    for (const [agentId, records] of byAgent) {
      await this.writeToAgentFile(agentId, records);
    }
    
    console.log(`[CaptureFileWriter] Flushed ${toWrite.length} records to files`);
  }

  /**
   * Write records to agent's daily file
   * 将记录写入 Agent 的每日文件
   */
  async writeToAgentFile(agentId, records) {
    const stream = await this.getOrCreateStream(agentId);
    
    for (const record of records) {
      const line = JSON.stringify(record) + '\n';
      stream.write(line);
    }
  }

  /**
   * Get or create write stream for agent
   * 获取或创建 Agent 的写入流
   */
  async getOrCreateStream(agentId) {
    const streamKey = `${agentId}-${this.currentDate}`;
    
    // Check if stream exists and is still valid
    // 检查流是否存在且仍然有效
    let stream = this.streams.get(streamKey);
    if (stream && !stream.destroyed) {
      return stream;
    }
    
    // Create agent directory if needed
    // 如果需要则创建 Agent 目录
    const agentDir = path.join(this.baseDir, agentId);
    await fs.promises.mkdir(agentDir, { recursive: true });
    
    // Create new stream (append mode)
    // 创建新流（追加模式）
    const filePath = path.join(agentDir, `${this.currentDate}.jsonl`);
    stream = fs.createWriteStream(filePath, { flags: 'a' });
    
    stream.on('error', (err) => {
      console.error(`[CaptureFileWriter] Write error for ${agentId}:`, err);
    });
    
    this.streams.set(streamKey, stream);
    console.log(`[CaptureFileWriter] Created stream for ${agentId} -> ${filePath}`);
    
    return stream;
  }

  /**
   * Get current date string (YYYY-MM-DD)
   * 获取当前日期字符串
   */
  getDateString() {
    return new Date().toISOString().split('T')[0];
  }

  /**
   * Start flush timer
   * 启动刷新定时器
   */
  startFlushTimer() {
    this.flushTimer = setInterval(async () => {
      try {
        await this.flush();
      } catch (err) {
        console.error('[CaptureFileWriter] Flush error:', err);
      }
    }, this.flushInterval);
  }

  /**
   * Start date rotation checker
   * 启动日期轮转检查器
   */
  startDateRotationChecker() {
    // Check every minute for date change
    // 每分钟检查日期变化
    setInterval(() => {
      const newDate = this.getDateString();
      if (newDate !== this.currentDate) {
        console.log(`[CaptureFileWriter] Date rotation: ${this.currentDate} -> ${newDate}`);
        
        this.currentDate = newDate;
        
        // Close old streams (will be recreated on next write)
        // 关闭旧流（下次写入时会重新创建）
        for (const [key, stream] of this.streams) {
          stream.end();
        }
        this.streams.clear();
      }
    }, 60000);
  }

  /**
   * Shutdown gracefully
   * 优雅关闭
   */
  async shutdown() {
    console.log('[CaptureFileWriter] Shutting down...');
    
    // Stop timer
    // 停止定时器
    if (this.flushTimer) {
      clearInterval(this.flushTimer);
    }
    
    // Final flush
    // 最终刷新
    await this.flush();
    
    // Close all streams
    // 关闭所有流
    for (const [key, stream] of this.streams) {
      stream.end();
    }
    this.streams.clear();
    
    console.log('[CaptureFileWriter] Shutdown complete');
  }
}

// Export singleton instance
// 导出单例实例
const captureFileWriter = new CaptureFileWriter();

module.exports = { CaptureFileWriter, captureFileWriter };
```

### Step 2: Integrate with LLM Proxy

Modify `src/services/llm-proxy.js` to use the file writer:
```javascript
// At the top of the file, add import:
// 在文件顶部添加导入：
const { captureFileWriter } = require('./capture-file-writer');

// In the constructor, initialize the file writer:
// 在构造函数中初始化文件写入器：
class LLMProxy {
  constructor(options) {
    // ... existing code ...
    
    // Initialize file writer
    // 初始化文件写入器
    captureFileWriter.initialize().catch(err => {
      console.error('[LLMProxy] Failed to initialize file writer:', err);
    });
  }

  // In your storeCapture method, add file write:
  // 在 storeCapture 方法中添加文件写入：
  storeCapture(agentId, capture) {
    // ... existing memory cache code ...
    
    // ✅ NEW: Also write to file (non-blocking)
    // ✅ 新增：同时写入文件（非阻塞）

  }
}
```

### Step 3: Add Graceful Shutdown

In your main server file (e.g., `src/server.js` or `src/index.js`):

```javascript
const { captureFileWriter } = require('./services/capture-file-writer');

// Graceful shutdown handler
// 优雅关闭处理器
process.on('SIGTERM', async () => {
  console.log('Received SIGTERM, shutting down gracefully...');
  await captureFileWriter.shutdown();
  process.exit(0);
});

process.on('SIGINT', async () => {
  console.log('Received SIGINT, shutting down gracefully...');

  process.exit(0);
});
```
